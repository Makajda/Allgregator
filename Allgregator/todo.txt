private int ComparisonReal(object x, object y) {
    if (x is FrameworkElement xe && y is FrameworkElement ye && xe.DataContext is IComparison xc && ye.DataContext is IComparison yc) {
        if (xc.Weight < yc.Weight) return -1;
        if (xc.Weight > yc.Weight) return 1;
    }

    return 0;
}
